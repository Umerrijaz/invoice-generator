        function handleLogoUpload(event) {

            const file = event.target.files[0];

            if (file && file.type.startsWith('image/')) {

                const reader = new FileReader();

                reader.onload = function(e) {

                    const logo = document.querySelector('.logo');

                    const logoText = document.getElementById('logoText');

                    

                    // Remove text and add image

                    logoText.style.display = 'none';

                    

                    // Remove existing image if any

                    const existingImg = logo.querySelector('img');

                    if (existingImg) {

                        existingImg.remove();

                    }

                    

                    // Create and add new image

                    const img = document.createElement('img');

                    img.src = e.target.result;

                    logo.appendChild(img);

                };

                reader.readAsDataURL(file);

            }

        }


        function addRow() {

            const table = document.getElementById('productsTable');

            const newRow = table.insertRow();

            newRow.innerHTML = `

                <td><input type="text" class="product-name product-name-input" placeholder="Product name"></td>

                <td><input type="number" class="quantity" value="1" min="1" onchange="calculateTotal()"></td>

                <td><input type="number" class="price" value="0" min="0" onchange="calculateTotal()"></td>

                <td><input type="number" class="item-total" value="0" readonly></td>

                <td><button class="btn btn-remove" onclick="removeRow(this)">×</button></td>

            `;

            calculateTotal();

        }


        function removeRow(button) {

            const row = button.parentElement.parentElement;

            row.remove();

            calculateTotal();

        }


        function calculateTotal() {

            const rows = document.querySelectorAll('#productsTable tr');

            let subtotal = 0;


            rows.forEach(row => {

                const quantity = parseFloat(row.querySelector('.quantity').value) || 0;

                const price = parseFloat(row.querySelector('.price').value) || 0;

                const itemTotal = quantity * price;

                row.querySelector('.item-total').value = itemTotal.toFixed(2);

                subtotal += itemTotal;

            });


            const deliveryCharges = parseFloat(document.getElementById('deliveryCharges').value) || 0;

            const discountAmount = parseFloat(document.getElementById('discount').value) || 0;

            const paymentMethod = document.getElementById('paymentMethod').value;

            

            // Calculate COD tax (4%) if payment method is Cash on Delivery

            let codTax = 0;

            const codTaxRow = document.getElementById('codTaxRow');

            

            if (paymentMethod === 'Cash on Delivery') {

                codTax = (subtotal + deliveryCharges - discountAmount) * 0.04;

                codTaxRow.style.display = 'flex';

                document.getElementById('codTax').textContent = codTax.toFixed(2);

            } else {

                codTaxRow.style.display = 'none';

                document.getElementById('codTax').textContent = '0.00';

            }

            

            const amountPaid = parseFloat(document.getElementById('amountPaid').value) || 0;

            

            const total = subtotal + deliveryCharges - discountAmount + codTax;

            const balanceDue = total - amountPaid;


            document.getElementById('subtotal').textContent = subtotal.toFixed(2);

            document.getElementById('total').textContent = total.toFixed(2);

            document.getElementById('balanceDue').textContent = balanceDue.toFixed(2);

        }


        async function generatePDF() {

            try {

                const { jsPDF } = window.jspdf;

                

                const pdfBtn = document.querySelector('.btn-pdf');

                const originalText = pdfBtn.textContent;

                pdfBtn.textContent = 'Generating PDF...';

                pdfBtn.disabled = true;

                

                const invoice = document.getElementById('invoice');

                

                // Convert all inputs and selects to display divs for PDF

                const inputs = invoice.querySelectorAll('input:not([type="button"]), select');

                const inputData = [];

                

                inputs.forEach(input => {

                    let value = '';

                    if (input.tagName === 'SELECT') {

                        value = input.options[input.selectedIndex].text;

                    } else {

                        value = input.value || input.placeholder || '';

                    }

                    

                    const div = document.createElement('div');

                    div.className = 'display-value';

                    div.textContent = value;

                    div.style.cssText = window.getComputedStyle(input).cssText;

                    div.style.border = 'none';

                    div.style.display = 'block';

                    

                    inputData.push({

                        input: input,

                        parent: input.parentNode,

                        next: input.nextSibling,

                        div: div

                    });

                    

                    input.parentNode.replaceChild(div, input);

                });

                

                // Hide buttons

                const buttons = invoice.querySelectorAll('.btn, .actions');

                buttons.forEach(btn => btn.style.display = 'none');

                

                // Wait for DOM update

                await new Promise(resolve => setTimeout(resolve, 200));

                

                // Capture with better quality

                const canvas = await html2canvas(invoice, {

                    scale: 3,

                    useCORS: true,

                    logging: false,

                    backgroundColor: '#ffffff',

                    width: invoice.offsetWidth,

                    height: invoice.offsetHeight,

                    windowWidth: invoice.offsetWidth,

                    windowHeight: invoice.offsetHeight

                });

                

                // Restore inputs

                inputData.forEach(data => {

                    data.parent.replaceChild(data.input, data.div);

                });

                

                // Show buttons

                buttons.forEach(btn => btn.style.display = '');

                

                // Create PDF

                const imgData = canvas.toDataURL('image/png', 1.0);

                const pdf = new jsPDF('p', 'mm', 'a4');

                

                const pdfWidth = pdf.internal.pageSize.getWidth();

                const pdfHeight = pdf.internal.pageSize.getHeight();

                const canvasWidth = canvas.width;

                const canvasHeight = canvas.height;

                const ratio = Math.min(pdfWidth / (canvasWidth / 3), pdfHeight / (canvasHeight / 3));

                

                const imgWidth = (canvasWidth / 3) * ratio;

                const imgHeight = (canvasHeight / 3) * ratio;

                const marginX = (pdfWidth - imgWidth) / 2;

                

                pdf.addImage(imgData, 'PNG', marginX, 10, imgWidth, imgHeight);

                

                const date = new Date().toLocaleDateString('en-PK').replace(/\//g, '-');

                pdf.save(`Invoice-${date}.pdf`);

                

                pdfBtn.textContent = originalText;

                pdfBtn.disabled = false;

                

            } catch (error) {

                console.error('PDF generation error:', error);

                alert('Error generating PDF. Please try using your browser print function (Ctrl+P or Cmd+P) and save as PDF.');

                

                const pdfBtn = document.querySelector('.btn-pdf');

                pdfBtn.textContent = 'Download PDF';

                pdfBtn.disabled = false;

            }

        }


        // Initial calculation

        calculateTotal();

    </script>
